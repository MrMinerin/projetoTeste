public class Calculadora
{
    public double som(double a, double b){
        double result = a + b;
        return result;
    }
    public double sub(double a, double b){
        double result = a - b;
        return result;
    }
    public double div(double a, double b){
        double result = a / b;
        return result;
    }
    public double mult(double a, double b){
        double result = a * b;
        return result;
    }
    public double elev(int a, int b){
        int u = 1;
        double result = 1;
        if (b < 0){
            for (int x = 0; x>b ;x-- ){
                u = u * a; 
            } 
            result = result/u;
        } else {
            for (int x = 0; x<b ;x++ ){
                u = u * a;
            } 
            result = result*u;
        }
        return result;
    }
